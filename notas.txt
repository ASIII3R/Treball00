        try {
            String filePath = "./prestecs.json";
            String content = new String(Files.readAllBytes(Paths.get(filePath)));

            // Convertir el contingut JSON a una llista o mapa d'objectes
            Gson gson = new Gson();
            Type listType = new TypeToken<List<Map<String, Object>>>() {}.getType();
            List<Map<String, Object>> prestecs = gson.fromJson(content, listType);

            // Iterar sobre els préstecs
            for (Map<String, Object> prestec : prestecs) {
                System.out.println("ID: " + prestec.get("id"));
                System.out.println("Data Préstec: " + prestec.get("dataPrestec"));
                System.out.println("Usuari ID: " + prestec.get("idUsuari"));
                System.out.println("---------------------------");
            }

        } catch (IOException e) {
            e.printStackTrace();
        }
